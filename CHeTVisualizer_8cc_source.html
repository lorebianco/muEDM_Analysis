8// Helper: Clipping algorithm to keep lines inside the detector box.
9// Moved to anonymous namespace to keep the header clean.
10bool ClipLineToBox(double x0, double y0, double z0, double ux, double uy, double uz, double xmin,
11 double xmax, double ymin, double ymax, double zmin, double zmax, double &tmin, double &tmax)
12{
13 // Use bounds provided by caller (e.g. -inf, +inf for line, or 0, L for segment)
14
15 auto clip = [&](double p0, double dp, double pmin, double pmax) -> bool
16 {
17 if(std::abs(dp) < 1e-12)
18 return (p0 >= pmin && p0 <= pmax);
19
20 double t1 = (pmin - p0) / dp;
21 double t2 = (pmax - p0) / dp;
22 if(t1 > t2)
23 std::swap(t1, t2);
24 tmin = std::max(tmin, t1);
25 tmax = std::min(tmax, t2);
26 return tmin <= tmax;
27 };
28
29 if(!clip(x0, ux, xmin, xmax))
30 return false;
31 if(!clip(y0, uy, ymin, ymax))
32 return false;
33 if(!clip(z0, uz, zmin, zmax))
34 return false;
35
36 return true;
37}

48void RenderTrack2D(const VisLineTrack &tr, TMultiGraph *mg_xy)
49{
50 TGraph *gxy = new TGraph();
51 // Use a large enough range so the line covers the detector view.
52 // The view limits are set by Draw2DCore based on geometry.
53 double range = 2000.0;
54 double pts[2] = { -range, range };
55
56 for(int i = 0; i < 2; i++)
57 {
58 if(std::abs(tr.ux) > 1e-6)
59 {
60 double t = (pts[i] - tr.x0) / tr.ux;
61 gxy->SetPoint(i, pts[i], tr.y0 + tr.uy * t);
62 }
63 else
64 {
65 gxy->SetPoint(0, tr.x0, -range);
66 gxy->SetPoint(1, tr.x0, range);
67 }
68 }
69 gxy->SetLineColor(tr.color);
70 gxy->SetLineWidth(tr.width);
71 gxy->SetLineStyle(tr.style);
72 mg_xy->Add(gxy, "L");
73}

220// --- PhiZ Projection (Phi vs Z) ---
221void RenderTrackPhiZ(const VisLineTrack &tr, TMultiGraph *mg_phiz, bool wrap_phi)
222{
223 int n_pts = 200;
224 TGraph *g = new TGraph();
225 double range = 1000.0;
226 int cnt = 0;
227 for(int i = 0; i < n_pts; ++i)
228 {
229 double t = -range + i * (2 * range) / (n_pts - 1);
230 double x = tr.x0 + tr.ux * t;
231 double y = tr.y0 + tr.uy * t;
232 double z = tr.z0 + tr.uz * t;
233
234 if(std::abs(z) > 400)
235 continue;
236
237 double phi = std::atan2(y, x);
238 if(wrap_phi && phi < 0)
239 phi += 2 * M_PI;
240 g->SetPoint(cnt++, phi, z);
241 }
242 g->SetLineColor(tr.color);
243 g->SetLineWidth(tr.width);
244 g->SetLineStyle(tr.style);
245 mg_phiz->Add(g, "L");
246}

292void RenderTrack3D(const VisLineTrack &tr_in, double bounds[6])
293{
294 VisLineTrack tr = tr_in;
295 if(tr.isLocalFrame)
296 {
297 CHeT::Config::ApplyTransformation(tr.x0, tr.y0, tr.z0);
298 CHeT::Config::ApplyRotation(tr.ux, tr.uy, tr.uz);
299 }
300
301 double tmin = -1e30;
302 double tmax = 1e30;
303 bool ok = ClipLineToBox(tr.x0, tr.y0, tr.z0, tr.ux, tr.uy, tr.uz, bounds[0], bounds[1],
304 bounds[2], bounds[3], bounds[4], bounds[5], tmin, tmax);
305 if(ok)
306 {
307 TPolyLine3D *lt = new TPolyLine3D(2);
308 lt->SetPoint(0, tr.z0 + tr.uz * tmin, tr.x0 + tr.ux * tmin, tr.y0 + tr.uy * tmin);
309 lt->SetPoint(1, tr.z0 + tr.uz * tmax, tr.x0 + tr.ux * tmax, tr.y0 + tr.uy * tmax);
310 lt->SetLineColor(tr.color);
311 lt->SetLineWidth(tr.width);
312 lt->SetLineStyle(tr.style);
313 lt->Draw("same");
314 }
315}

422 vx.reserve(500);

480 sf.clear();
481 }
482 sz.push_back(vz[i]);
483 sf.push_back(cf);
484 }

487 if(!sz.empty())
488 {
489 TGraph *hf = new TGraph();
490 int nn = sz.size();
491 for(int j = 0; j < nn; ++j)
492 hf->SetPoint(j, sf[j] + d_phi_w, sz[j]);
493 for(int j = 0; j < nn; ++j)
494 hf->SetPoint(nn + j, sf[nn - 1 - j] - d_phi_w, sz[nn - 1 - j]);
495
496 hf->SetFillColorAlpha(p.color, 0.3);
497 hf->SetLineWidth(0);
498 mg_phiz->Add(hf, "F");
499

501 lf->SetLineColor(p.color);
502 lf->SetLineWidth(2);
503 mg_phiz->Add(lf, "L");
504 }
505 }
506
507 // --- Draw Tracks (All Projections) ---
508 if(trackDrawer)

510
511 // --- Draw Extra Points (XY) ---
512 for(const auto &pt : extraPoints)
513 {
514 TGraph *gp = new TGraph(1);
515 gp->SetPoint(0, pt.x, pt.y);
516 gp->SetMarkerColor(pt.color);
517 gp->SetMarkerStyle(pt.markerStyle);
518 gp->SetMarkerSize(pt.size);
519 mg_xy->Add(gp, "P");

528 for(size_t i = 0; i < inters.size(); ++i)

531 gi_xy->SetMarkerSize(0.8);

533 mg_xy->Add(gi_xy, "P");

535 // ZX (x vs z)
536 TGraph *gi_zx = new TGraph();
537 for(size_t i = 0; i < inters.size(); ++i)
538 gi_zx->SetPoint(i, inters[i].x_loc, inters[i].z_loc);
539 gi_zx->SetMarkerStyle(20);
540 gi_zx->SetMarkerSize(0.8);
541 gi_zx->SetMarkerColor(kBlack);
542 mg_zx->Add(gi_zx, "P");
543

651void Draw3DCore(const std::vector<int> &hit_ids, const std::vector<VisPoint3D> &points,
652 bool drawSkeleton, const std::function<void(double[6])> &trackDrawer)
653{
654 // gStyle->SetCanvasPreferGL(kTRUE);
655
656 TCanvas *c_3d = (TCanvas *)gROOT->FindObject("c_3d");
657 if(!c_3d)
658 c_3d = new TCanvas("c_3d", "Detector 3D View", 1200, 800);
659 else
660 c_3d->Clear();
661
662 // Calculate dynamic bounding box based on geometry and transformation
663 double min_x_phys = 1e9, max_x_phys = -1e9;
664 double min_y_phys = 1e9, max_y_phys = -1e9;
665 double min_z_phys = 1e9, max_z_phys = -1e9;
666
667 auto cylinders = CHeT::Config::GetCylinders();
668 double max_R = 0;
669 for(const auto &cyl : cylinders)
670 {
671 if(cyl.outer.radius > max_R)
672 max_R = cyl.outer.radius;
673 }
674 if(max_R < 1.0)
675 max_R = 100.0; // Fallback
676
677 double corners[8][3]
678 = { { max_R, max_R, CHeT::Config::L_HALF }, { max_R, -max_R, CHeT::Config::L_HALF },
679 { -max_R, max_R, CHeT::Config::L_HALF }, { -max_R, -max_R, CHeT::Config::L_HALF },
680 { max_R, max_R, -CHeT::Config::L_HALF }, { max_R, -max_R, -CHeT::Config::L_HALF },
681 { -max_R, max_R, -CHeT::Config::L_HALF }, { -max_R, -max_R, -CHeT::Config::L_HALF } };
682
683 for(int i = 0; i < 8; ++i)
684 {
685 double x = corners[i][0];
686 double y = corners[i][1];
687 double z = corners[i][2];
688 CHeT::Config::ApplyTransformation(x, y, z);
689
690 if(x < min_x_phys)
691 min_x_phys = x;
692 if(x > max_x_phys)
693 max_x_phys = x;
694 if(y < min_y_phys)
695 min_y_phys = y;
696 if(y > max_y_phys)
697 max_y_phys = y;
698 if(z < min_z_phys)
699 min_z_phys = z;
700 if(z > max_z_phys)
701 max_z_phys = z;
702 }
703
704 double margin = 20.0;
705 min_x_phys -= margin;
706 max_x_phys += margin;
707 min_y_phys -= margin;
708 max_y_phys += margin;
709 min_z_phys -= margin;
710 max_z_phys += margin;
711
712 // Force aspect ratio to be isotropic (Cube) only if rotated
713 double rx, ry, rz;
714 CHeT::Config::GetRotation(rx, ry, rz);
715 bool isRotated = (std::abs(rx) > 1e-9 || std::abs(ry) > 1e-9 || std::abs(rz) > 1e-9);
716
717 if(isRotated)
718 {
719 double dx = max_x_phys - min_x_phys;
720 double dy = max_y_phys - min_y_phys;
721 double dz = max_z_phys - min_z_phys;
722 double max_dim = std::max({ dx, dy, dz });
723
724 double cx = (min_x_phys + max_x_phys) / 2.0;
725 double cy = (min_y_phys + max_y_phys) / 2.0;
726 double cz = (min_z_phys + max_z_phys) / 2.0;
727
728 min_x_phys = cx - max_dim / 2.0;
729 max_x_phys = cx + max_dim / 2.0;
730 min_y_phys = cy - max_dim / 2.0;
731 max_y_phys = cy + max_dim / 2.0;
732 min_z_phys = cz - max_dim / 2.0;
733 max_z_phys = cz + max_dim / 2.0;
734 }
735
736 double bounds[6] = { min_x_phys, max_x_phys, min_y_phys, max_y_phys, min_z_phys, max_z_phys };
737
738 // ROOT Frame: X->Z_phys, Y->X_phys, Z->Y_phys
739 TH3F *h_frame = new TH3F("h_frame", "; Z [mm]; X [mm]; Y [mm]", 1, min_z_phys, max_z_phys, 1,
740 min_x_phys, max_x_phys, 1, min_y_phys, max_y_phys);
741 h_frame->SetDirectory(0); // Detach from directory to prevent ROOT auto-deletion issues
742 h_frame->SetStats(0);
743 h_frame->GetXaxis()->SetTitleOffset(1.5);
744 h_frame->GetYaxis()->SetTitleOffset(1.5);
745 h_frame->GetZaxis()->SetTitleOffset(1.);
746 h_frame->Draw();
747
748 // --- Draw Skeleton (Inactive fibers) ---
749 if(drawSkeleton)
750 {
751 auto cylinders = CHeT::Config::GetCylinders();
752 for(const auto &cyl : cylinders)
753 {
754 const CHeT::Config::LayerConfig *layers[2] = { &cyl.inner, &cyl.outer };
755 for(int l = 0; l < 2; ++l)
756 {
757 for(int b = 0; b < layers[l]->nBundles; ++b)
758 {
759 int global_id = CHeT::Config::GetGlobalIdFromGeometry(cyl.id, l, b);
760 CHeT::Config::FiberProp p = CHeT::Config::GetFiberProp(global_id);
761
762 // Ghost fiber
763 TPolyLine3D *bg_f = new TPolyLine3D(10);
764 for(int i = 0; i < 10; ++i)
765 {
766 double z = -CHeT::Config::L_HALF + i * (2.0 * CHeT::Config::L_HALF / 9.0);
767 double a = (z + CHeT::Config::L_HALF) / (2.0 * CHeT::Config::L_HALF);
768 double ph = p.phi0 + p.dir * a * M_PI;
769
770 double x3 = p.r * std::cos(ph);
771 double y3 = p.r * std::sin(ph);
772 double z3 = z;
773 CHeT::Config::ApplyTransformation(x3, y3, z3);
774
775 // Map to ROOT: X_root=z, Y_root=x, Z_root=y
776 bg_f->SetPoint(i, z3, x3, y3);
777 }
778 bg_f->SetLineColorAlpha(p.color, 0.1); // Very transparent
779 bg_f->Draw("same");
780 }
781 }
782 }
783 }
784
785 // --- Draw Hits (Active fibers) ---
786 for(int id : hit_ids)
787 {
788 if(id < 0)
789 continue;
790 CHeT::Config::FiberProp p = CHeT::Config::GetFiberProp(id);
791
792 TPolyLine3D *fl = new TPolyLine3D(50); // More detailed for active ones
793 for(int i = 0; i < 50; ++i)
794 {
795 double z = -CHeT::Config::L_HALF + i * (2.0 * CHeT::Config::L_HALF / 49.0);
796 double a = (z + CHeT::Config::L_HALF) / (2.0 * CHeT::Config::L_HALF);
797 double ph = p.phi0 + p.dir * a * M_PI;
798
799 double x3 = p.r * std::cos(ph);
800 double y3 = p.r * std::sin(ph);
801 double z3 = z;
802 CHeT::Config::ApplyTransformation(x3, y3, z3);
803
804 // Map to ROOT: X_root=z, Y_root=x, Z_root=y
805 fl->SetPoint(i, z3, x3, y3);
806 }
807 fl->SetLineColor(p.color);
808 fl->SetLineWidth(3);
809 fl->Draw("same");
810 }
811
812 // --- Draw Tracks ---
813 if(trackDrawer)
814 trackDrawer(bounds);
815
816 // --- Draw 3D Points ---
817 for(const auto &pt_in : points)
818 {
819 VisPoint3D pt = pt_in;
820 if(pt.isLocalFrame)
821 {
822 CHeT::Config::ApplyTransformation(pt.x, pt.y, pt.z);
823 }
824
825 TPolyMarker3D *pm = new TPolyMarker3D(1);
826 // Note: Mapping Physics (x,y,z) to ROOT 3D Frame (Z, X, Y)
827 // ROOT X axis = Physics Z
828 // ROOT Y axis = Physics X
829 // ROOT Z axis = Physics Y
830 pm->SetPoint(0, pt.z, pt.x, pt.y);
831
832 pm->SetMarkerColor(pt.color);
833 pm->SetMarkerStyle(pt.markerStyle);
834 pm->SetMarkerSize(pt.size);
835 pm->Draw("same");
836 }
837
838 c_3d->Update();
839}
