8// Helper: Clipping algorithm to keep lines inside the detector box.
9// Moved to anonymous namespace to keep the header clean.
10bool ClipLineToBox(double x0, double y0, double z0, double ux, double uy, double uz, double xmin,
11 double xmax, double ymin, double ymax, double zmin, double zmax, double &tmin, double &tmax)
12{
13 tmin = -1e30;
14 tmax = 1e30;
15
16 auto clip = [&](double p0, double dp, double pmin, double pmax) -> bool
17 {
18 if(std::abs(dp) < 1e-12)
19 return (p0 >= pmin && p0 <= pmax);
20
21 double t1 = (pmin - p0) / dp;
22 double t2 = (pmax - p0) / dp;
23 if(t1 > t2)
24 std::swap(t1, t2);
25 tmin = std::max(tmin, t1);
26 tmax = std::min(tmax, t2);
27 return tmin <= tmax;
28 };
29
30 if(!clip(x0, ux, xmin, xmax))
31 return false;
32 if(!clip(y0, uy, ymin, ymax))
33 return false;
34 if(!clip(z0, uz, zmin, zmax))
35 return false;
36
37 return true;
38}

301 const std::vector<VisPoint3D> &points, bool drawSkeleton)
302{

304
305 TCanvas *c_3d = (TCanvas *)gROOT->FindObject("c_3d");
306 if(!c_3d)
307 c_3d = new TCanvas("c_3d", "Detector 3D View", 1200, 800);
308 else
309 c_3d->Clear();
310
311 // Calculate dynamic bounding box based on geometry and transformation
312 double min_x_phys = 1e9, max_x_phys = -1e9;
313 double min_y_phys = 1e9, max_y_phys = -1e9;
314 double min_z_phys = 1e9, max_z_phys = -1e9;
315
316 auto cylinders = CHeT::Config::GetCylinders();
317 double max_R = 0;
318 for(const auto &cyl : cylinders)
319 {
320 if(cyl.outer.radius > max_R)
321 max_R = cyl.outer.radius;
322 }
323 if(max_R < 1.0)
324 max_R = 100.0; // Fallback
325
326 double corners[8][3]
327 = { { max_R, max_R, CHeT::Config::L_HALF }, { max_R, -max_R, CHeT::Config::L_HALF },
328 { -max_R, max_R, CHeT::Config::L_HALF }, { -max_R, -max_R, CHeT::Config::L_HALF },
329 { max_R, max_R, -CHeT::Config::L_HALF }, { max_R, -max_R, -CHeT::Config::L_HALF },
330 { -max_R, max_R, -CHeT::Config::L_HALF }, { -max_R, -max_R, -CHeT::Config::L_HALF } };
331
332 for(int i = 0; i < 8; ++i)
333 {
334 double x = corners[i][0];
335 double y = corners[i][1];
336 double z = corners[i][2];
337 CHeT::Config::ApplyTransformation(x, y, z);
338
339 if(x < min_x_phys)
340 min_x_phys = x;
341 if(x > max_x_phys)
342 max_x_phys = x;
343 if(y < min_y_phys)
344 min_y_phys = y;
345 if(y > max_y_phys)
346 max_y_phys = y;
347 if(z < min_z_phys)
348 min_z_phys = z;
349 if(z > max_z_phys)
350 max_z_phys = z;
351 }
352
353 double margin = 20.0;
354 min_x_phys -= margin;
355 max_x_phys += margin;
356 min_y_phys -= margin;
357 max_y_phys += margin;
358 min_z_phys -= margin;
359 max_z_phys += margin;
360
361 // Force aspect ratio to be isotropic (Cube) only if rotated
362 double rx, ry, rz;
363 CHeT::Config::GetRotation(rx, ry, rz);
364 bool isRotated = (std::abs(rx) > 1e-9 || std::abs(ry) > 1e-9 || std::abs(rz) > 1e-9);
365
366 if(isRotated)
367 {
368 double dx = max_x_phys - min_x_phys;
369 double dy = max_y_phys - min_y_phys;
370 double dz = max_z_phys - min_z_phys;
371 double max_dim = std::max({ dx, dy, dz });
372
373 double cx = (min_x_phys + max_x_phys) / 2.0;
374 double cy = (min_y_phys + max_y_phys) / 2.0;
375 double cz = (min_z_phys + max_z_phys) / 2.0;
376
377 min_x_phys = cx - max_dim / 2.0;
378 max_x_phys = cx + max_dim / 2.0;
379 min_y_phys = cy - max_dim / 2.0;
380 max_y_phys = cy + max_dim / 2.0;
381 min_z_phys = cz - max_dim / 2.0;
382 max_z_phys = cz + max_dim / 2.0;
383 }
384
385 // ROOT Frame: X->Z_phys, Y->X_phys, Z->Y_phys
386 TH3F *h_frame = new TH3F("h_frame", "; Z [mm]; X [mm]; Y [mm]", 1, min_z_phys, max_z_phys, 1,
387 min_x_phys, max_x_phys, 1, min_y_phys, max_y_phys);
388 h_frame->SetDirectory(0); // Detach from directory to prevent ROOT auto-deletion issues
389 h_frame->SetStats(0);
390 h_frame->GetXaxis()->SetTitleOffset(1.5);
391 h_frame->GetYaxis()->SetTitleOffset(1.5);
392 h_frame->GetZaxis()->SetTitleOffset(1.);
393 h_frame->Draw();
394
395 // --- Draw Skeleton (Inactive fibers) ---
396 if(drawSkeleton)
397 {
398 auto cylinders = CHeT::Config::GetCylinders();
399 int b_global_idx = 0;
400
401 for(const auto &cyl : cylinders)
402 {
403 const CHeT::Config::LayerConfig *layers[2] = { &cyl.inner, &cyl.outer };
404 for(int l = 0; l < 2; ++l)
405 {
406 for(int b = 0; b < layers[l]->nBundles; ++b)
407 {
408 CHeT::Config::FiberProp p = CHeT::Config::GetFiberProp(b_global_idx + b);
409
410 // Ghost fiber
411 TPolyLine3D *bg_f = new TPolyLine3D(10);
412 for(int i = 0; i < 10; ++i)
413 {
414 double z = -CHeT::Config::L_HALF + i * (2.0 * CHeT::Config::L_HALF / 9.0);
415 double a = (z + CHeT::Config::L_HALF) / (2.0 * CHeT::Config::L_HALF);
416 double ph = p.phi0 + p.dir * a * M_PI;
417
418 double x3 = p.r * std::cos(ph);
419 double y3 = p.r * std::sin(ph);
420 double z3 = z;
421 CHeT::Config::ApplyTransformation(x3, y3, z3);
422
423 // Map to ROOT: X_root=z, Y_root=x, Z_root=y
424 bg_f->SetPoint(i, z3, x3, y3);
425 }
426 bg_f->SetLineColorAlpha(p.color, 0.1); // Very transparent
427 bg_f->Draw("same");
428 }
429 b_global_idx += layers[l]->nBundles;
430 }
431 }
432 }
433
434 // --- Draw Hits (Active fibers) ---
435 for(int id : hit_ids)
436 {
437 if(id < 0)
438 continue;
439 CHeT::Config::FiberProp p = CHeT::Config::GetFiberProp(id);
440
441 TPolyLine3D *fl = new TPolyLine3D(50); // More detailed for active ones
442 for(int i = 0; i < 50; ++i)
443 {
444 double z = -CHeT::Config::L_HALF + i * (2.0 * CHeT::Config::L_HALF / 49.0);
445 double a = (z + CHeT::Config::L_HALF) / (2.0 * CHeT::Config::L_HALF);
446 double ph = p.phi0 + p.dir * a * M_PI;
447
448 double x3 = p.r * std::cos(ph);
449 double y3 = p.r * std::sin(ph);
450 double z3 = z;
451 CHeT::Config::ApplyTransformation(x3, y3, z3);
452
453 // Map to ROOT: X_root=z, Y_root=x, Z_root=y
454 fl->SetPoint(i, z3, x3, y3);
455 }
456 fl->SetLineColor(p.color);
457 fl->SetLineWidth(3);
458 fl->Draw("same");
459 }
460
461 // --- Draw Tracks ---
462 for(const auto &tr_in : tracks)
463 {
464 VisLineTrack tr = tr_in;
465 if(tr.isLocalFrame)
466 {
467 CHeT::Config::ApplyTransformation(tr.x0, tr.y0, tr.z0);
468 CHeT::Config::ApplyRotation(tr.ux, tr.uy, tr.uz);
469 }
470
471 double tmin, tmax;
472 // Use the internal helper function
473 bool ok
474 = ClipLineToBox(tr.x0, tr.y0, tr.z0, tr.ux, tr.uy, tr.uz, min_x_phys, max_x_phys, // X
475 min_y_phys, max_y_phys, // Y
476 min_z_phys, max_z_phys, // Z
477 tmin, tmax);
478
479 if(ok)
480 {
481 TPolyLine3D *lt = new TPolyLine3D(2);
482 lt->SetPoint(0, tr.z0 + tr.uz * tmin, tr.x0 + tr.ux * tmin, tr.y0 + tr.uy * tmin);
483 lt->SetPoint(1, tr.z0 + tr.uz * tmax, tr.x0 + tr.ux * tmax, tr.y0 + tr.uy * tmax);
484 lt->SetLineColor(tr.color);
485 lt->SetLineWidth(tr.width);
486 lt->SetLineStyle(tr.style);
487 lt->Draw("same");
488 }
489 }
490
491 // --- Draw 3D Points ---
492 for(const auto &pt_in : points)
493 {
494 VisPoint3D pt = pt_in;
495 if(pt.isLocalFrame)
496 {
497 CHeT::Config::ApplyTransformation(pt.x, pt.y, pt.z);
498 }
499
500 TPolyMarker3D *pm = new TPolyMarker3D(1);
501 // Note: Mapping Physics (x,y,z) to ROOT 3D Frame (Z, X, Y)
502 // ROOT X axis = Physics Z
503 // ROOT Y axis = Physics X
504 // ROOT Z axis = Physics Y
505 pm->SetPoint(0, pt.z, pt.x, pt.y);
506
507 pm->SetMarkerColor(pt.color);
508 pm->SetMarkerStyle(pt.markerStyle);
509 pm->SetMarkerSize(pt.size);
510 pm->Draw("same");
511 }
512
513 c_3d->Update();
514}
