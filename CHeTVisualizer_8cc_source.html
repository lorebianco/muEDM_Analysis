7// Helper: Clipping algorithm to keep lines inside the detector box.
8// Moved to anonymous namespace to keep the header clean.
9bool ClipLineToBox(double x0, double y0, double z0, double ux, double uy,
10 double uz, double xmin, double xmax, double ymin, double ymax, double zmin,
11 double zmax, double &tmin, double &tmax)
12{
13 tmin = -1e30;
14 tmax = 1e30;
15
16 auto clip = [&](double p0, double dp, double pmin, double pmax) -> bool
17 {
18 if(std::abs(dp) < 1e-12)
19 return (p0 >= pmin && p0 <= pmax);
20
21 double t1 = (pmin - p0) / dp;
22 double t2 = (pmax - p0) / dp;
23 if(t1 > t2)
24 std::swap(t1, t2);
25 tmin = std::max(tmin, t1);
26 tmax = std::min(tmax, t2);
27 return tmin <= tmax;
28 };
29
30 if(!clip(x0, ux, xmin, xmax))
31 return false;
32 if(!clip(y0, uy, ymin, ymax))
33 return false;
34 if(!clip(z0, uz, zmin, zmax))
35 return false;
36
37 return true;
38}

307void Draw3D(const std::vector<int> &hit_ids,
308 const std::vector<VisLineTrack> &tracks,
309 const std::vector<VisPoint3D> &points, bool drawSkeleton)
310{
311 // gStyle->SetCanvasPreferGL(kTRUE);
312
313 TCanvas *c_3d = (TCanvas *)gROOT->FindObject("c_3d");
314 if(!c_3d)
315 c_3d = new TCanvas("c_3d", "Detector 3D View", 1200, 800);
316 else
317 c_3d->Clear();
318
319 // Frame limits
320 const double z_lim = 190.0;
321 const double r_lim = 35.0;
322
323 TH3F *h_frame = new TH3F("h_frame", "; Z [mm]; X [mm]; Y [mm]", 1, -z_lim,
324 z_lim, 1, -r_lim, r_lim, 1, -r_lim, r_lim);
325 h_frame->SetDirectory(
326 0); // Detach from directory to prevent ROOT auto-deletion issues
327 h_frame->SetStats(0);
328 h_frame->GetXaxis()->SetTitleOffset(1.5);
329 h_frame->GetYaxis()->SetTitleOffset(1.5);
330 h_frame->GetZaxis()->SetTitleOffset(1.);
331 h_frame->Draw();
332
333 // --- Draw Skeleton (Inactive fibers) ---
334 if(drawSkeleton)
335 {
336 auto cylinders = CHeT::Config::GetCylinders();
337 int b_global_idx = 0;
338
339 for(const auto &cyl : cylinders)
340 {
341 const CHeT::Config::LayerConfig *layers[2]
342 = { &cyl.inner, &cyl.outer };
343 for(int l = 0; l < 2; ++l)
344 {
345 for(int b = 0; b < layers[l]->nBundles; ++b)
346 {
347 CHeT::Config::FiberProp p
348 = CHeT::Config::GetFiberProp(b_global_idx + b);
349
350 // Ghost fiber
351 TPolyLine3D *bg_f = new TPolyLine3D(10);
352 for(int i = 0; i < 10; ++i)
353 {
354 double z = -CHeT::Config::L_HALF
355 + i * (2.0 * CHeT::Config::L_HALF / 9.0);
356 double a = (z + CHeT::Config::L_HALF)
357 / (2.0 * CHeT::Config::L_HALF);
358 double ph = p.phi0 + p.dir * a * M_PI;
359 bg_f->SetPoint(
360 i, z, p.r * std::cos(ph), p.r * std::sin(ph));
361 }
362 bg_f->SetLineColorAlpha(p.color, 0.1); // Very transparent
363 bg_f->Draw("same");
364 }
365 b_global_idx += layers[l]->nBundles;
366 }
367 }
368 }
369
370 // --- Draw Hits (Active fibers) ---
371 for(int id : hit_ids)
372 {
373 if(id < 0)
374 continue;
375 CHeT::Config::FiberProp p = CHeT::Config::GetFiberProp(id);
376
377 TPolyLine3D *fl = new TPolyLine3D(50); // More detailed for active ones
378 for(int i = 0; i < 50; ++i)
379 {
380 double z = -CHeT::Config::L_HALF
381 + i * (2.0 * CHeT::Config::L_HALF / 49.0);
382 double a
383 = (z + CHeT::Config::L_HALF) / (2.0 * CHeT::Config::L_HALF);
384 double ph = p.phi0 + p.dir * a * M_PI;
385 fl->SetPoint(i, z, p.r * std::cos(ph), p.r * std::sin(ph));
386 }
387 fl->SetLineColor(p.color);
388 fl->SetLineWidth(3);
389 fl->Draw("same");
390 }
391
392 // --- Draw Tracks ---
393 for(const auto &tr : tracks)
394 {
395 double tmin, tmax;
396 // Use the internal helper function
397 bool ok = ClipLineToBox(tr.x0, tr.y0, tr.z0, tr.ux, tr.uy, tr.uz,
398 -r_lim, r_lim, // X
399 -r_lim, r_lim, // Y
400 -z_lim, z_lim, // Z
401 tmin, tmax);
402
403 if(ok)
404 {
405 TPolyLine3D *lt = new TPolyLine3D(2);
406 lt->SetPoint(0, tr.z0 + tr.uz * tmin, tr.x0 + tr.ux * tmin,
407 tr.y0 + tr.uy * tmin);
408 lt->SetPoint(1, tr.z0 + tr.uz * tmax, tr.x0 + tr.ux * tmax,
409 tr.y0 + tr.uy * tmax);
410 lt->SetLineColor(tr.color);
411 lt->SetLineWidth(tr.width);
412 lt->SetLineStyle(tr.style);
413 lt->Draw("same");
414 }
415 }
416
417 // --- Draw 3D Points ---
418 for(const auto &pt : points)
419 {
420 TPolyMarker3D *pm = new TPolyMarker3D(1);
421 // Note: Mapping Physics (x,y,z) to ROOT 3D Frame (Z, X, Y)
422 // ROOT X axis = Physics Z
423 // ROOT Y axis = Physics X
424 // ROOT Z axis = Physics Y
425 pm->SetPoint(0, pt.z, pt.x, pt.y);
426
427 pm->SetMarkerColor(pt.color);
428 pm->SetMarkerStyle(pt.markerStyle);
429 pm->SetMarkerSize(pt.size);
430 pm->Draw("same");
431 }
432
433 c_3d->Update();
434}
