46ROOT::RDF::RNode Reader::GetEstimators()
47{
48 // Capture parameters for lambdas
49 double tMin = fToaMin;
50 double tMax = fToaMax;
51 unsigned int totMin = fTotMin;
52 unsigned int totMax = fTotMax;
53
54 // 1. ToA Corrections (Alignment w.r.t. FD00)
55 // FD00 is the reference: only LSB -> ns conversion
56 auto df_corr = fHeadNode.Define("FD00_corrToA",
57 [](const RVecUI &toa) { return RVecD(toa) / 2.0; }, { "FD00_ToA" });
58
59 // Other boards (FD01..FD03) are aligned using fine timestamp
60 for(const string &id : { "01", "02", "03" })
61 {
62 df_corr = df_corr.Define("FD" + id + "_corrToA",
63 [](const RVecUI &toa, Double_t t, Double_t t0)
64 {
65 // Correction: ToA/2 + fine timestamp difference (ps -> ns?)
66 // Note: original code had * 1e3, check units!
67 return RVecD(toa) / 2.0 + (t - t0) * 1e3;
68 },
69 { "FD" + id + "_ToA", "FD" + id + "_fine_tstamp",
70 "FD00_fine_tstamp" });
71 }
72
73 // 2. Loop Board -> Selection -> Bundle Mapping
74 auto df_geo = df_corr;
75 vector<string> toa_cols, tot_cols, cyl_cols, lay_cols, bund_cols;
76
77 // Map string ID -> numeric ID for geometry
78 map<string, int> board_id_map
79 = { { "00", 0 }, { "01", 1 }, { "02", 2 }, { "03", 3 } };
80
81 for(const auto &[id_str, id_num] : board_id_map)
82 {
83 string id = id_str; // "00", "01"...
84 int b_id = id_num;
85
86 string toa_n = "FD" + id + "_ToA_sel";
87 string tot_n = "FD" + id + "_ToT_sel";
88 string cyl_n = "FD" + id + "_cyl_sel";
89 string lay_n = "FD" + id + "_lay_sel";
90 string bund_n = "FD" + id + "_bund_sel";
91
92 toa_cols.push_back(toa_n);
93 tot_cols.push_back(tot_n);
94 cyl_cols.push_back(cyl_n);
95 lay_cols.push_back(lay_n);
96 bund_cols.push_back(bund_n);
97
98 // Time cuts on ToA and ToT
99 df_geo = df_geo.Define(toa_n,
100 [=](const RVecD &toa, const RVecUS &tot)
101 {
102 return toa[(toa > tMin) && (toa < tMax) && (tot > totMin)
103 && (tot < totMax)];
104 },
105 { "FD" + id + "_corrToA", "FD" + id + "_ToT" });
106
107 df_geo = df_geo.Define(tot_n,
108 [=](const RVecD &toa, const RVecUS &tot)
109 {
110 return tot[(toa > tMin) && (toa < tMax) && (tot > totMin)
111 && (tot < totMax)];
112 },
113 { "FD" + id + "_corrToA", "FD" + id + "_ToT" });
114
115 // Bundle ID Calculation (Geometry)
116 df_geo = df_geo.Define(bund_n,
117 [=](const RVecD &toa, const RVecUS &tot, const RVecUC &chans)
118 {
119 RVecI bunds;
120 auto mask = (toa > tMin) && (toa < tMax) && (tot > totMin)
121 && (tot < totMax);
122 auto sel_chans = chans[mask];
123 for(auto c : sel_chans)
124 {
125 int gid = CHeT::Config::GetGlobalBundleId(b_id, (int)c);
126 bunds.push_back(gid);
127 }
128 return bunds;
129 },
130 { "FD" + id + "_corrToA", "FD" + id + "_ToT",
131 "FD" + id + "_channel" });
132
133 // Derive Cylinder and Layer from Global IDs
134 df_geo = df_geo.Define(cyl_n,
135 [](const RVecI &bunds)
136 {
137 RVecI cyls;
138 for(auto b : bunds)
139 cyls.push_back((b >= 0)
140 ? CHeT::Config::GetFiberProp(b).cylinderId
141 : -1);
142 return cyls;
143 },
144 { bund_n });
145
146 df_geo = df_geo.Define(lay_n,
147 [](const RVecI &bunds)
148 {
149 RVecI lays;
150 for(auto b : bunds)
151 lays.push_back(
152 (b >= 0) ? CHeT::Config::GetFiberProp(b).layerId : -1);
153 return lays;
154 },
155 { bund_n });
156 }
157
158 // 3. Global Concatenation (All Boards -> Single Event Vector)
159 auto df_final
160 = df_geo
161 .Define(
162 "All_Cyl",
163 [](const RVecI &c0, const RVecI &c1, const RVecI &c2,
164 const RVecI &c3) {
165 return Concatenate(
166 c0, Concatenate(c1, Concatenate(c2, c3)));
167 },
168 cyl_cols)
169 .Define(
170 "All_Lay",
171 [](const RVecI &l0, const RVecI &l1, const RVecI &l2,
172 const RVecI &l3) {
173 return Concatenate(
174 l0, Concatenate(l1, Concatenate(l2, l3)));
175 },
176 lay_cols)
177 .Define(
178 "All_Bundle",
179 [](const RVecI &b0, const RVecI &b1, const RVecI &b2,
180 const RVecI &b3) {
181 return Concatenate(
182 b0, Concatenate(b1, Concatenate(b2, b3)));
183 },
184 bund_cols)
185
186 // Counts and Event-Level variables
187 .Define("nHits_Total", "(double)All_Cyl.size()")
188 .Define("nHits_Cyl0", "Sum(All_Cyl == 0)")
189 .Define("nHits_Cyl1", "Sum(All_Cyl == 1)")
190 .Define("nHits_Cyl0_Inner", "Sum(All_Cyl == 0 && All_Lay == 0)")
191 .Define("nHits_Cyl0_Outer", "Sum(All_Cyl == 0 && All_Lay == 1)")
192 .Define("nHits_Cyl1_Inner", "Sum(All_Cyl == 1 && All_Lay == 0)")
193 .Define("nHits_Cyl1_Outer", "Sum(All_Cyl == 1 && All_Lay == 1)")
194 .Define(
195 "FirstToA",
196 [](const RVecD &t0, const RVecD &t1, const RVecD &t2,
197 const RVecD &t3)
198 {
199 auto all = Concatenate(
200 t0, Concatenate(t1, Concatenate(t2, t3)));
201 return all.empty()
202 ? std::numeric_limits<Double_t>::infinity()
203 : Min(all);
204 },
205 toa_cols)
206 .Define(
207 "SumToT",
208 [](const RVecUS &t0, const RVecUS &t1, const RVecUS &t2,
209 const RVecUS &t3)
210 {
211 auto all = Concatenate(
212 t0, Concatenate(t1, Concatenate(t2, t3)));
213 return (Double_t)Sum(all);
214 },
215 tot_cols);
216
217 return df_final;
218}
