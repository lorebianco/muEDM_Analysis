64ROOT::RDF::RNode Reader::GetEstimators()
65{
66 // Capture parameters for lambdas
67 double tMin = fToaMin;
68 double tMax = fToaMax;
69 unsigned int totMin = fTotMin;
70 unsigned int totMax = fTotMax;
71
72 // Capture filters
73 auto enabledBoards = fEnabledBoards;
74 auto enabledCylinders = fEnabledCylinders;
75 auto enabledLayers = fEnabledLayers;
76 auto enabledGeometries = fEnabledGeometries;
77
78 // 1. ToA Corrections (Alignment w.r.t. FD00)
79 // FD00 is the reference: only LSB -> ns conversion
80 auto df_corr = fHeadNode.Define(
81 "FD00_corrToA", [](const RVecUI &toa) { return RVecD(toa) / 2.0; }, { "FD00_ToA" });
82
83 // Other boards (FD01..FD03) are aligned using fine timestamp
84 for(const string &id : { "01", "02", "03" })
85 {
86 df_corr = df_corr.Define("FD" + id + "_corrToA",
87 [](const RVecUI &toa, Double_t t, Double_t t0)
88 {
89 // Correction: ToA/2 + fine timestamp difference (ps -> ns?)
90 // Note: original code had * 1e3, check units!
91 return RVecD(toa) / 2.0 + (t - t0) * 1e3;
92 },
93 { "FD" + id + "_ToA", "FD" + id + "_fine_tstamp", "FD00_fine_tstamp" });
94 }
95
96 // 2. Loop Board -> Selection -> Bundle Mapping
97 auto df_geo = df_corr;
98 vector<string> toa_cols, tot_cols, cyl_cols, lay_cols, bund_cols;
99
100 // Map string ID -> numeric ID for geometry
101 map<string, int> board_id_map = { { "00", 0 }, { "01", 1 }, { "02", 2 }, { "03", 3 } };
102
103 for(const auto &[id_str, id_num] : board_id_map)
104 {
105 string id = id_str; // "00", "01"...
106 int b_id = id_num;
107
108 string toa_n = "FD" + id + "_ToA_sel";
109 string tot_n = "FD" + id + "_ToT_sel";
110 string cyl_n = "FD" + id + "_cyl_sel";
111 string lay_n = "FD" + id + "_lay_sel";
112 string bund_n = "FD" + id + "_bund_sel";
113
114 toa_cols.push_back(toa_n);
115 tot_cols.push_back(tot_n);
116 cyl_cols.push_back(cyl_n);
117 lay_cols.push_back(lay_n);
118 bund_cols.push_back(bund_n);
119
120 // Check Board Filtering
121 bool isBoardEnabled = true;
122 if(!enabledBoards.empty())
123 {
124 if(std::find(enabledBoards.begin(), enabledBoards.end(), b_id) == enabledBoards.end())
125 {
126 isBoardEnabled = false;
127 }
128 }
129
130 if(!isBoardEnabled)
131 {
132 // Define empty columns if board is disabled
133 // We use specific types to match the expected types in Concatenate
134 df_geo = df_geo.Define(toa_n, []() { return RVecD {}; })
135 .Define(tot_n, []() { return RVecUS {}; })
136 .Define(cyl_n, []() { return RVecI {}; })
137 .Define(lay_n, []() { return RVecI {}; })
138 .Define(bund_n, []() { return RVecI {}; });
139 continue;
140 }
141
142 // If board is enabled, we first calculate everything into temporary
143 // columns, then apply cylinder/layer filtering.
144 string toa_tmp = toa_n + "_tmp";
145 string tot_tmp = tot_n + "_tmp";
146 string cyl_tmp = cyl_n + "_tmp";
147 string lay_tmp = lay_n + "_tmp";
148 string bund_tmp = bund_n + "_tmp";
149
150 // Time cuts on ToA and ToT
151 df_geo = df_geo.Define(toa_tmp,
152 [=](const RVecD &toa, const RVecUS &tot)
153 { return toa[(toa > tMin) && (toa < tMax) && (tot > totMin) && (tot < totMax)]; },
154 { "FD" + id + "_corrToA", "FD" + id + "_ToT" });
155
156 df_geo = df_geo.Define(tot_tmp,
157 [=](const RVecD &toa, const RVecUS &tot)
158 { return tot[(toa > tMin) && (toa < tMax) && (tot > totMin) && (tot < totMax)]; },
159 { "FD" + id + "_corrToA", "FD" + id + "_ToT" });
160
161 // Bundle ID Calculation (Geometry)
162 df_geo = df_geo.Define(bund_tmp,
163 [=](const RVecD &toa, const RVecUS &tot, const RVecUC &chans)
164 {
165 RVecI bunds;
166 auto mask = (toa > tMin) && (toa < tMax) && (tot > totMin) && (tot < totMax);
167 auto sel_chans = chans[mask];
168 for(auto c : sel_chans)
169 {
170 int gid = CHeT::Config::GetGlobalBundleId(b_id, (int)c);
171 bunds.push_back(gid);
172 }
173 return bunds;
174 },
175 { "FD" + id + "_corrToA", "FD" + id + "_ToT", "FD" + id + "_channel" });
176
177 // Derive Cylinder and Layer from Global IDs
178 df_geo = df_geo.Define(cyl_tmp,
179 [](const RVecI &bunds)
180 {
181 RVecI cyls;
182 for(auto b : bunds)
183 cyls.push_back((b >= 0) ? CHeT::Config::GetFiberProp(b).cylinderId : -1);
184 return cyls;
185 },
186 { bund_tmp });
187
188 df_geo = df_geo.Define(lay_tmp,
189 [](const RVecI &bunds)
190 {
191 RVecI lays;
192 for(auto b : bunds)
193 lays.push_back((b >= 0) ? CHeT::Config::GetFiberProp(b).layerId : -1);
194 return lays;
195 },
196 { bund_tmp });
197
198 // Create Filter Mask based on Cylinder and Layer
199 string mask_n = "FD" + id + "_geo_mask";
200 df_geo = df_geo.Define(mask_n,
201 [enabledCylinders, enabledLayers, enabledGeometries](
202 const RVecI &cyls, const RVecI &lays)
203 {
204 // By default, keep everything (mask=1)
205 RVec<int> mask(cyls.size(), 1);
206
207 // If no filters are set, return all 1s
208 if(enabledCylinders.empty() && enabledLayers.empty() && enabledGeometries.empty())
209 {
210 return mask;
211 }
212
213 for(size_t i = 0; i < cyls.size(); ++i)
214 {
215 // Check Cylinder
216 if(!enabledCylinders.empty())
217 {
218 bool found = false;
219 for(auto c : enabledCylinders)
220 if(c == cyls[i])
221 {
222 found = true;
223 break;
224 }
225 if(!found)
226 {
227 mask[i] = 0;
228 continue;
229 } // Optimization: if fail cyl, fail hit
230 }
231
232 // Check Layer
233 if(!enabledLayers.empty())
234 {
235 bool found = false;
236 for(auto l : enabledLayers)
237 if(l == lays[i])
238 {
239 found = true;
240 break;
241 }
242 if(!found)
243 mask[i] = 0;
244 }
245
246 // Check Specific Geometries (Cyl, Lay)
247 if(!enabledGeometries.empty())
248 {
249 bool found = false;
250 for(const auto &geo : enabledGeometries)
251 {
252 if(geo.first == cyls[i] && geo.second == lays[i])
253 {
254 found = true;
255 break;
256 }
257 }
258 if(!found)
259 mask[i] = 0;
260 }
261 }
262 return mask;
263 },
264 { cyl_tmp, lay_tmp });
265
266 // Apply Mask to final columns
267 // We use explicit lambdas to ensure type correctness
268 df_geo = df_geo
269 .Define(toa_n, [](const RVecD &v, const RVecI &m) { return v[m]; },
270 { toa_tmp, mask_n })
271 .Define(tot_n, [](const RVecUS &v, const RVecI &m) { return v[m]; },
272 { tot_tmp, mask_n })
273 .Define(cyl_n, [](const RVecI &v, const RVecI &m) { return v[m]; },
274 { cyl_tmp, mask_n })
275 .Define(lay_n, [](const RVecI &v, const RVecI &m) { return v[m]; },
276 { lay_tmp, mask_n })
277 .Define(bund_n, [](const RVecI &v, const RVecI &m) { return v[m]; },
278 { bund_tmp, mask_n });
279 }
280
281 // 3. Global Concatenation (All Boards -> Single Event Vector)
282 auto df_final
283 = df_geo
284 .Define(
285 "All_Cyl",
286 [](const RVecI &c0, const RVecI &c1, const RVecI &c2, const RVecI &c3)
287 { return Concatenate(c0, Concatenate(c1, Concatenate(c2, c3))); },
288 cyl_cols)
289 .Define(
290 "All_Lay",
291 [](const RVecI &l0, const RVecI &l1, const RVecI &l2, const RVecI &l3)
292 { return Concatenate(l0, Concatenate(l1, Concatenate(l2, l3))); },
293 lay_cols)
294 .Define(
295 "All_Bundle",
296 [](const RVecI &b0, const RVecI &b1, const RVecI &b2, const RVecI &b3)
297 { return Concatenate(b0, Concatenate(b1, Concatenate(b2, b3))); },
298 bund_cols)
299
300 // Counts and Event-Level variables
301 .Define("nHits_Total", "(double)All_Cyl.size()")
302 .Define("nHits_Cyl0", "Sum(All_Cyl == 0)")
303 .Define("nHits_Cyl1", "Sum(All_Cyl == 1)")
304 .Define("nHits_Cyl0_Inner", "Sum(All_Cyl == 0 && All_Lay == 0)")
305 .Define("nHits_Cyl0_Outer", "Sum(All_Cyl == 0 && All_Lay == 1)")
306 .Define("nHits_Cyl1_Inner", "Sum(All_Cyl == 1 && All_Lay == 0)")
307 .Define("nHits_Cyl1_Outer", "Sum(All_Cyl == 1 && All_Lay == 1)")
308 .Define(
309 "FirstToA",
310 [](const RVecD &t0, const RVecD &t1, const RVecD &t2, const RVecD &t3)
311 {
312 auto all = Concatenate(t0, Concatenate(t1, Concatenate(t2, t3)));
313 return all.empty() ? std::numeric_limits<Double_t>::infinity() : Min(all);
314 },
315 toa_cols)
316 .Define(
317 "SumToT",
318 [](const RVecUS &t0, const RVecUS &t1, const RVecUS &t2, const RVecUS &t3)
319 {
320 auto all = Concatenate(t0, Concatenate(t1, Concatenate(t2, t3)));
321 return (Double_t)Sum(all);
322 },
323 tot_cols);
324
325 return df_final;
326}
