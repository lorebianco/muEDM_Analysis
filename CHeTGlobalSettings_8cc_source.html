126void SetRotation(double rx, double ry, double rz)
127{
128 g_RotX = rx;
129 g_RotY = ry;
130 g_RotZ = rz;
131 g_IsRotated
132 = (std::abs(rx) > 1e-9 || std::abs(ry) > 1e-9 || std::abs(rz) > 1e-9);
133
134 if(g_IsRotated)
135 {
136 double cx = std::cos(rx), sx = std::sin(rx);
137 double cy = std::cos(ry), sy = std::sin(ry);
138 double cz = std::cos(rz), sz = std::sin(rz);
139
140 // Calculate Rotation Matrix Columns by transforming basis vectors
141 auto rotate = [&](double x, double y, double z, double &ox, double &oy,
142 double &oz)
143 {
144 // Rx
145 double y1 = y * cx - z * sx;
146 double z1 = y * sx + z * cx;
147 double x1 = x;
148 // Ry
149 double z2 = z1 * cy - x1 * sy;
150 double x2 = z1 * sy + x1 * cy;
151 double y2 = y1;
152 // Rz
153 double x3 = x2 * cz - y2 * sz;
154 double y3 = x2 * sz + y2 * cz;
155 double z3 = z2;
156 ox = x3;
157 oy = y3;
158 oz = z3;
159 };
160
161 // Col 0
162 rotate(1, 0, 0, g_R[0][0], g_R[1][0], g_R[2][0]);
163 // Col 1
164 rotate(0, 1, 0, g_R[0][1], g_R[1][1], g_R[2][1]);
165 // Col 2
166 rotate(0, 0, 1, g_R[0][2], g_R[1][2], g_R[2][2]);
167 }
168 else
169 {
170 // Identity
171 g_R[0][0] = 1;
172 g_R[0][1] = 0;
173 g_R[0][2] = 0;
174 g_R[1][0] = 0;
175 g_R[1][1] = 1;
176 g_R[1][2] = 0;
177 g_R[2][0] = 0;
178 g_R[2][1] = 0;
179 g_R[2][2] = 1;
180 }
181}

204void ApplyRotation(double &x, double &y, double &z)
205{
206 if(!g_IsRotated)
207 return;
208
209 double xn = g_R[0][0] * x + g_R[0][1] * y + g_R[0][2] * z;
210 double yn = g_R[1][0] * x + g_R[1][1] * y + g_R[1][2] * z;
211 double zn = g_R[2][0] * x + g_R[2][1] * y + g_R[2][2] * z;
212
213 x = xn;
214 y = yn;
215 z = zn;
216}

218void ApplyInverseRotation(double &x, double &y, double &z)
219{
220 if(!g_IsRotated)
221 return;
222
223 // Transpose multiplication
224 double xn = g_R[0][0] * x + g_R[1][0] * y + g_R[2][0] * z;
225 double yn = g_R[0][1] * x + g_R[1][1] * y + g_R[2][1] * z;
226 double zn = g_R[0][2] * x + g_R[1][2] * y + g_R[2][2] * z;
227
228 x = xn;
229 y = yn;
230 z = zn;
231}

233void ApplyTransformation(double &x, double &y, double &z)
234{
235 // 1. Rotation (Local -> Aligned Global)
236 ApplyRotation(x, y, z);
237
238 // 2. Translation (Aligned Global -> Global)
239 x += g_TransX;
240 y += g_TransY;
241 z += g_TransZ;
242}

244void ApplyInverseTransformation(double &x, double &y, double &z)
245{
246 // 1. Inverse Translation (Global -> Aligned Global)
247 x -= g_TransX;
248 y -= g_TransY;
249 z -= g_TransZ;
250
251 // 2. Inverse Rotation (Aligned Global -> Local)
252 ApplyInverseRotation(x, y, z);
253}

292FiberProp GetFiberProp(int b_id)
293{
294 const auto &cylinders = GetCylinders();
295 int offset = 0;
296
297 for(const auto &cyl : cylinders)
298 {
299 // Check Inner
300 if(b_id < offset + cyl.inner.nBundles)
301 {
302 int b_loc = b_id - offset;
303 return { cyl.id, 0, cyl.inner.radius,
304 wrap0_2pi(2.0 * M_PI * (-b_loc) / cyl.inner.nBundles
305 + cyl.inner.phiOffset),
306 cyl.inner.direction, cyl.inner.color };
307 }
308 offset += cyl.inner.nBundles;
309
310 // Check Outer
311 if(b_id < offset + cyl.outer.nBundles)
312 {
313 int b_loc = b_id - offset;
314 return { cyl.id, 1, cyl.outer.radius,
315 wrap0_2pi(2.0 * M_PI * (b_loc) / cyl.outer.nBundles
316 + cyl.outer.phiOffset),
317 cyl.outer.direction, cyl.outer.color };
318 }
319 offset += cyl.outer.nBundles;
320 }
321 // Return invalid prop
322 return { -1, -1, 0, 0, 0, 0 };

325int GetGlobalBundleId(int board_id, int channel_id)
326{
327 auto boardMapIt = BOARD_MAPS.find(board_id);
328 if(boardMapIt == BOARD_MAPS.end())
329 return -1;
330
331 auto chanIt = boardMapIt->second.find(channel_id);
332 if(chanIt == boardMapIt->second.end())
333 return -1;
334
335 int local_val = chanIt->second;
336
337 // Apply offset
338 int boardOffset = GetBoardGlobalOffset(board_id);
339 if(boardOffset == -1)
340 return -1;
341
342 return local_val + boardOffset;
343}

345std::vector<BundlesIntersection> FindIntersections(
346 const std::vector<int> &hit_ids)
347{
348 std::vector<BundlesIntersection> out;
349 // Optimization: reserve memory if hits are many, though tricky to guess how
350 // many intersections
351
352 for(size_t i = 0; i < hit_ids.size(); ++i)
353 {
354 for(size_t j = i + 1; j < hit_ids.size(); ++j)
355 {
356 FiberProp p1 = GetFiberProp(hit_ids[i]);
357 FiberProp p2 = GetFiberProp(hit_ids[j]);
358
359 // Intersection logic: Same cylinder, different layers
360 if(p1.cylinderId == p2.cylinderId && p1.cylinderId != -1
361 && p1.layerId != p2.layerId)
362 {
363 double dphi0 = p2.phi0 - p1.phi0;
364 double ddir = p1.dir - p2.dir;
365
366 // Typical case: +1 vs -1 -> diff is 2 or -2.
367 // Loop over possible periodic intersections
368 for(int k = -2; k <= 2; ++k)
369 {
370 // Avoid division by zero
371 if(std::abs(ddir) < 1e-9)
372 continue;
373
374 double alpha = (dphi0 + 2.0 * k * M_PI) / (ddir * M_PI);
375
376 if(alpha >= 0.0 && alpha <= 1.0)
377 {
378 double zi = alpha * (2.0 * L_HALF) - L_HALF;
379 double ph = p1.phi0 + p1.dir * alpha * M_PI;
380
381 // Z Thickness Calculation
382 // DeltaZ depends on Bundle Width and Helix slope.
383 double deltaZ = (BUNDLE_WIDTH / p1.r) * (L_HALF / M_PI);
384
385 // Calculate local coordinates
386 double x_loc = p1.r * std::cos(ph);
387 double y_loc = p1.r * std::sin(ph);
388 double z_loc = zi;
389
390 // Apply global rotation (to copies)
391 double x_rot = x_loc;
392 double y_rot = y_loc;
393 double z_rot = z_loc;
394 ApplyTransformation(x_rot, y_rot, z_rot);
395
396 out.push_back({ z_rot, x_rot, y_rot, z_loc, x_loc,
397 y_loc, p1.cylinderId, deltaZ });
398 }
399 }
400 }
401 }
402 }
403 return out;
404}

408int GetGlobalIdFromGeometry(int cyl_id, int layer_id, int layer_idx)
409{
410 const auto &cylinders = GetCylinders();
411 int global_offset = 0;
412
413 for(const auto &cyl : cylinders)
414 {
415 if(cyl.id == cyl_id)
416 {
417 // Check limits
418 int limit
419 = (layer_id == 0) ? cyl.inner.nBundles : cyl.outer.nBundles;
420
421 if(layer_idx < 0 || layer_idx >= limit)
422 {
423 std::cerr << " [ERROR] Index " << layer_idx
424 << " out of range for Cylinder " << cyl_id
425 << " Layer " << layer_id << "!\n";
426 return -1;
427 }
428
429 // Calculate ID: Offset + Local Base + Index
430 int local_base = (layer_id == 0) ? 0 : cyl.inner.nBundles;
431 return global_offset + local_base + layer_idx;
432 }
433 // Accumulate offset if not current cylinder
434 global_offset += (cyl.inner.nBundles + cyl.outer.nBundles);
435 }
436 return -1; // Cylinder not found
437}

439void PrintBundleMapping(int global_id)
440{
441 if(global_id < 0)
442 return;
443
444 std::cout << "\n >>> INFO BUNDLE ID: " << global_id << " <<<\n";
445
446 // A. Check Hardware Mapping
447 bool found_hw = false;
448 for(auto const &[board, map] : BOARD_MAPS)
449 {
450 int off = GetBoardGlobalOffset(board);
451 if(off < 0)
452 continue;
453
454 // Reverse lookup: value == global_id - offset
455 int target_local = global_id - off;
456
457 for(auto const &[chan, val] : map)
458 {
459 if(val == target_local)
460 {
461 std::cout << " [HARDWARE] Board: " << board
462 << " | Channel: " << chan << "\n";
463 found_hw = true;
464 break;
465 }
466 }
467 if(found_hw)
468 break;
469 }
470 if(!found_hw)
471 std::cout << " [HARDWARE] Not mapped on any board.\n";
472
473 // B. Check Geometry Mapping
474 const auto &cylinders = GetCylinders();
475 int current_offset = 0;
476 bool found_geo = false;
477
478 for(const auto &cyl : cylinders)
479 {
480 int tot = cyl.inner.nBundles + cyl.outer.nBundles;
481 if(global_id >= current_offset && global_id < current_offset + tot)
482 {
483 int local = global_id - current_offset;
484 bool isInner = (local < cyl.inner.nBundles);
485 int layerIdx = isInner ? local : (local - cyl.inner.nBundles);
486
487 std::cout << " [GEOMETRY] Cylinder: " << cyl.id
488 << " | Layer: " << (isInner ? "INNER" : "OUTER")
489 << " | Index: " << layerIdx << "\n";
490 found_geo = true;
491 break;
492 }
493 current_offset += tot;
494 }
495
496 if(!found_geo)
497 std::cout << " [GEOMETRY] ID out of geometric range.\n";
498}
