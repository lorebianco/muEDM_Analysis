108FiberProp GetFiberProp(int b_id)
109{
110 const auto &cylinders = GetCylinders();
111 int offset = 0;
112
113 for(const auto &cyl : cylinders)
114 {
115 // Check Inner
116 if(b_id < offset + cyl.inner.nBundles)
117 {
118 int b_loc = b_id - offset;
119 return { cyl.id, 0, cyl.inner.radius,
120 wrap0_2pi(2.0 * M_PI * (-b_loc) / cyl.inner.nBundles
121 + cyl.inner.phiOffset),
122 cyl.inner.direction, cyl.inner.color };
123 }
124 offset += cyl.inner.nBundles;
125
126 // Check Outer
127 if(b_id < offset + cyl.outer.nBundles)
128 {
129 int b_loc = b_id - offset;
130 return { cyl.id, 1, cyl.outer.radius,
131 wrap0_2pi(2.0 * M_PI * (b_loc) / cyl.outer.nBundles
132 + cyl.outer.phiOffset),
133 cyl.outer.direction, cyl.outer.color };
134 }
135 offset += cyl.outer.nBundles;
136 }
137 // Return invalid prop
138 return { -1, -1, 0, 0, 0, 0 };
139}

141int GetGlobalBundleId(int board_id, int channel_id)
142{
143 auto boardMapIt = BOARD_MAPS.find(board_id);
144 if(boardMapIt == BOARD_MAPS.end())
145 return -1;
146
147 auto chanIt = boardMapIt->second.find(channel_id);
148 if(chanIt == boardMapIt->second.end())
149 return -1;
150
151 int local_val = chanIt->second;
152
153 // Apply offset
154 int boardOffset = GetBoardGlobalOffset(board_id);
155 if(boardOffset == -1)
156 return -1;
157
158 return local_val + boardOffset;
159}

161std::vector<BundlesIntersection> FindIntersections(
162 const std::vector<int> &hit_ids)
163{
164 std::vector<BundlesIntersection> out;
165 // Optimization: reserve memory if hits are many, though tricky to guess how
166 // many intersections
167
168 for(size_t i = 0; i < hit_ids.size(); ++i)
169 {
170 for(size_t j = i + 1; j < hit_ids.size(); ++j)
171 {
172 FiberProp p1 = GetFiberProp(hit_ids[i]);
173 FiberProp p2 = GetFiberProp(hit_ids[j]);
174
175 // Intersection logic: Same cylinder, different layers
176 if(p1.cylinderId == p2.cylinderId && p1.cylinderId != -1
177 && p1.layerId != p2.layerId)
178 {
179 double dphi0 = p2.phi0 - p1.phi0;
180 double ddir = p1.dir - p2.dir;
181
182 // Typical case: +1 vs -1 -> diff is 2 or -2.
183 // Loop over possible periodic intersections
184 for(int k = -2; k <= 2; ++k)
185 {
186 // Avoid division by zero
187 if(std::abs(ddir) < 1e-9)
188 continue;
189
190 double alpha = (dphi0 + 2.0 * k * M_PI) / (ddir * M_PI);
191
192 if(alpha >= 0.0 && alpha <= 1.0)
193 {
194 double zi = alpha * (2.0 * L_HALF) - L_HALF;
195 double ph = p1.phi0 + p1.dir * alpha * M_PI;
196
197 // Z Thickness Calculation
198 // DeltaZ depends on Bundle Width and Helix slope.
199 double deltaZ = (BUNDLE_WIDTH / p1.r) * (L_HALF / M_PI);
200
201 out.push_back({ zi, p1.r * std::cos(ph),
202 p1.r * std::sin(ph), p1.cylinderId, deltaZ });
203 }
204 }
205 }
206 }
207 }
208 return out;
209}

213int GetGlobalIdFromGeometry(int cyl_id, int layer_id, int layer_idx)
214{
215 const auto &cylinders = GetCylinders();
216 int global_offset = 0;
217
218 for(const auto &cyl : cylinders)
219 {
220 if(cyl.id == cyl_id)
221 {
222 // Check limits
223 int limit
224 = (layer_id == 0) ? cyl.inner.nBundles : cyl.outer.nBundles;
225
226 if(layer_idx < 0 || layer_idx >= limit)
227 {
228 std::cerr << " [ERROR] Index " << layer_idx
229 << " out of range for Cylinder " << cyl_id
230 << " Layer " << layer_id << "!\n";
231 return -1;
232 }
233
234 // Calculate ID: Offset + Local Base + Index
235 int local_base = (layer_id == 0) ? 0 : cyl.inner.nBundles;
236 return global_offset + local_base + layer_idx;
237 }
238 // Accumulate offset if not current cylinder
239 global_offset += (cyl.inner.nBundles + cyl.outer.nBundles);
240 }
241 return -1; // Cylinder not found
242}

244void PrintBundleMapping(int global_id)
245{
246 if(global_id < 0)
247 return;
248
249 std::cout << "\n >>> INFO BUNDLE ID: " << global_id << " <<<\n";
250
251 // A. Check Hardware Mapping
252 bool found_hw = false;
253 for(auto const &[board, map] : BOARD_MAPS)
254 {
255 int off = GetBoardGlobalOffset(board);
256 if(off < 0)
257 continue;
258
259 // Reverse lookup: value == global_id - offset
260 int target_local = global_id - off;
261
262 for(auto const &[chan, val] : map)
263 {
264 if(val == target_local)
265 {
266 std::cout << " [HARDWARE] Board: " << board
267 << " | Channel: " << chan << "\n";
268 found_hw = true;
269 break;
270 }
271 }
272 if(found_hw)
273 break;
274 }
275 if(!found_hw)
276 std::cout << " [HARDWARE] Not mapped on any board.\n";
277
278 // B. Check Geometry Mapping
279 const auto &cylinders = GetCylinders();
280 int current_offset = 0;
281 bool found_geo = false;
282
283 for(const auto &cyl : cylinders)
284 {
285 int tot = cyl.inner.nBundles + cyl.outer.nBundles;
286 if(global_id >= current_offset && global_id < current_offset + tot)
287 {
288 int local = global_id - current_offset;
289 bool isInner = (local < cyl.inner.nBundles);
290 int layerIdx = isInner ? local : (local - cyl.inner.nBundles);
291
292 std::cout << " [GEOMETRY] Cylinder: " << cyl.id
293 << " | Layer: " << (isInner ? "INNER" : "OUTER")
294 << " | Index: " << layerIdx << "\n";
295 found_geo = true;
296 break;
297 }
298 current_offset += tot;
299 }
300
301 if(!found_geo)
302 std::cout << " [GEOMETRY] ID out of geometric range.\n";
303}
